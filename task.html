1.To print the factorial of the maximum digit in the number 
            var num=prompt("Enter number "),prod=1;
            var max= num[0];
            for(i in num){
                if(num[i]>max){
                    max=num[i];
                }
            }
            for(var i=1;i<=max;i++){
                    prod=prod*i;
            }
            console.log("Maximum Factorial is: "+prod);
